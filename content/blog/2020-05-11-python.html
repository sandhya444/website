library(reticulate)

nt <- "CAGCCCAATCAGGCTCTACTGCCACTAAACTTACGCAGGATATATTTACGCCGACGTACT"

def kmer(sequence, k):
    counts={}
    find = len(sequence)-k+1 
    for i in range(find):
        rf=sequence[i:i+k]
        if rf not in counts:
            counts[rf]=0
        counts[rf]+=1
    return counts

kmer(r.nt, 3)
